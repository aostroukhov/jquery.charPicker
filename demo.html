jQuery Plugin · v1.2.0

jquery.charPicker

Вставка спецсимволов и эмодзи в input/textarea. Поиск с умной коррекцией раскладки EN↔RU. BEM-стили, никаких зависимостей кроме jQuery.

01

Быстрый старт

html 
<!-- 1. Подключить стили -->
<link rel="stylesheet" href="jquery.charPicker.css">

<!-- 2. Разметка -->
<input type="text" id="field" placeholder="Введите текст…">

<!-- 3. Подключить jQuery и плагин -->
<script src="jquery-3.7.1.min.js"></script>
<script src="jquery.charPicker.js"></script>

<!-- 4. Инициализация -->
<script>
  $('#field').charPicker();
</script>
💡
После инициализации рядом с полем появится кнопка ☺. Клик открывает попап с поиском и 10 панелями символов. Клик по символу вставляет его в позицию курсора.

03

API

инициализация
// Defaults
$('#f').charPicker();

// С опциями
$('#f').charPicker({ panels: ['math'] });

// Несколько полей
$('input, textarea').charPicker();
методы
// Открыть попап
$('#f').charPicker('open');

// Закрыть попап
$('#f').charPicker('close');

// Удалить плагин
$('#f').charPicker('destroy');
прямой доступ
// Экземпляр CharPicker
var inst = $('#f')
  .data('cp-instance');

if (inst) {
  inst.open();
  inst.close();
  inst.destroy();
}
МетодОписаниеПовтор
charPicker(options) Инициализация. Повторный вызов на том же элементе — игнорируется. нет эффекта
charPicker('open') Открывает попап, фокусирует поиск, пересчитывает позицию. нет эффекта
charPicker('close') Закрывает попап, скрывает тултип, обновляет aria-expanded. нет эффекта
charPicker('destroy') Снимает обработчики, удаляет DOM-элементы, возвращает input в исходное место. ошибка
04

Опции

javascript
$('#field').charPicker({
  recentMax : 32,         // макс. символов в «Недавних»
  position  : 'auto',    // 'auto' | 'top' | 'bottom'
  panels    : null,      // null = все; ['smileys'] = только эти; [] = ни одной
  onOpen    : null,      // function(), this = el
  onClose   : null,      // function(), this = el
  onSelect  : null       // function(char, name), this = el
});
recentMax
number32
Максимум символов в панели «Недавние». При превышении самые старые вытесняются новыми.
position
'auto' | 'top' | 'bottom''auto'
auto — вниз если достаточно места, вверх если снизу тесно. top/bottom — принудительно. Всегда прижимается к правому краю экрана если не умещается слева.
panels
Array | nullnull
null — показать все группы. ['smileys','math'] — только эти. [] — ни одной (только «Недавние» если есть). Панель recent управляется автоматически.
onOpen
function | nullnull
Вызывается при открытии попапа. this — DOM-элемент input/textarea.
onClose
function | nullnull
Вызывается при закрытии попапа. this — DOM-элемент input/textarea.
onSelect
function | nullnull
Вызывается после вставки символа. Аргументы: function(char, name). this.value уже содержит обновлённое значение.
onSelect — пример
$('#editor').charPicker({
  onSelect: function(char, name) {
    // char — вставленный символ:  '😀'
    // name — его составное имя:   'grinning smile улыбка'
    // this  — DOM-элемент input/textarea
    // this.value уже обновлено
    console.log('Вставлен:', char, '—', name);
  }
});

06

Поиск и умная раскладка

Каждый символ имеет составное двуязычное имя. Поиск ищет подстроку без учёта регистра, debounce 200 мс. Главная особенность — автоматическая коррекция раскладки клавиатуры.

формат имён символов
😀  'grinning smile улыбка'
→   'вправо right arrow'
©   'авторское право copyright'
€   'евро euro'
À   'a с акутом a with grave'

Алгоритм коррекции раскладки

Хотел «сердце», набрал в EN
cthlwt → сердце
Нашёл: ❤️ 🧡 💛 💚 💙
Хотел «arrow», набрал в RU
фккщц → arrow
Нашёл: → ← ↑ ↓ ↔ ↕ …
Хотел «улыбка», набрал в EN
ekmirrf → улыбка
Нашёл: 😀 😊 😄 😁 …
Правильный ввод «heart»
heart → heart
Нашёл напрямую: ❤️ 🧡 💕 …
реализация · _queryVariants
// Таблица QWERTY → ЙЦУКЕН (ГОСТ Р 52659–2006)
var _LAT_TO_CYR = { 'q':'й', 'w':'ц', 'e':'у', 'r':'к', /* … */ };

// Обратная таблица строится автоинверсией — не дублируется вручную
var _CYR_TO_LAT = (function() {
  var map = {};
  for (var lat in _LAT_TO_CYR) map[_LAT_TO_CYR[lat]] = lat;
  return map;
}());

// Варианты запроса: [оригинал, lat→cyr, cyr→lat]
function _queryVariants(q) {
  var variants = [q];
  var asCyr = _convertLayout(q, _LAT_TO_CYR);
  if (asCyr !== q) variants.push(asCyr);
  var asLat = _convertLayout(q, _CYR_TO_LAT);
  if (asLat !== q && asLat !== asCyr) variants.push(asLat);
  return variants; // 1–3 варианта, без дубликатов
}

09

Безопасность

🔒
XSS исключён. Все символы вставляются через document.createTextNode() — HTML-теги экранируются как текст. innerHTML с пользовательскими данными не используется нигде.
🚫
Нет eval. В плагине нет ни одного вызова eval(), new Function() или динамической компиляции кода. Все DOM-узлы создаются через jQuery с константными строками.
⚡
Изоляция событий. Каждый экземпляр использует уникальный jQuery namespace (click.cp.cp4k9m2x). destroy() снимает только свои обработчики — соседние экземпляры не затрагиваются.
✏️
Экранирование кавычек. Символы вроде ' ' " « заданы через \uXXXX — синтаксические конфликты с JS-строками исключены при парсинге файла.
createTextNode — защита от XSS
// В _makeGrid: символ → DOM через createTextNode
$ch[0].appendChild(document.createTextNode(item.c));

// Даже если item.c = '<script>alert(1)</script>' —
// браузер отобразит это как текст, не выполнит

// В insertAtCursor: вставка через .value, не innerHTML
el.value = el.value.slice(0, start) + text + el.value.slice(end);
